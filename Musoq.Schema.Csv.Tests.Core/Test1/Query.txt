table CsvFileTable {
	fileName 'string',
	intValue 'decimal',
	stringValue 'string'
};

couple #csv.file with table CsvFileTable as SourceOfRows;

with ReadedFiles as (
	select fileName, intValue, stringValue from SourceOfRows('E:\CsvTests\file1.csv', ',', true, 0)
	union all (fileName, intValue, stringValue)
	select fileName, intValue, stringValue from SourceOfRows('E:\CsvTests\file2.csv', ',', true, 0)
)
select * from ReadedFiles
